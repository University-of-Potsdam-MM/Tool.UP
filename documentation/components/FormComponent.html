  generateInserId()
  {
    function sortNumber(a,b)
    {
        return a - b;
    }

  let splitedids = [];
  let ids = [];
    this.appsfromWebservice.forEach(element =>
    {
      splitedids.push(element.id.split("-"));
    });
    splitedids.forEach(element =>
    {
      ids.push((element[1]));
    });

    ids.sort(sortNumber);
    let insertID = Number.parseInt(ids[ids.length-1], 10) + 1;
    return "application-" + insertID;
  }
